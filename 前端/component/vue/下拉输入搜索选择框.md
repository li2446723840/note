```vue
<template>
  <div class="custom-select-component">
    <el-select v-model="bindValue" value-key="code" multiple :disabled="disabled" placeholder="请选择客户商圈" :size="size" popper-class="custom-popover-style" @focus="focusSelect">
      <div class="custom-search-box">
        <el-input v-model="busKeywords" placeholder="请输入想要查找的商圈" :size="size" class="search-input" prefix-icon="el-icon-search" clearable @clear="clearSearch">
          <el-button slot="append" :size="size" @click="searchCondition = busKeywords">查询</el-button>
        </el-input>
      </div>
      <el-option v-for="item in filterOptions" :key="item.code" :label="item.name" :value="item">
        <span>
          <span class="icon-box">
            <i class="el-icon-check"></i>
          </span>
          {{ item.name }}</span
        >
      </el-option>
    </el-select>
  </div>
</template>

<script>
export default {
  name: "businessSelect",
  props: {
    value: {
      type: Array,
      default() {
        return [];
      }
    },
    disabled: {
      type: Boolean,
      default: false
    },
    groupCircleOpts: {
      type: Array,
      default() {
        return [];
      }
    },
    size: {
      type: String,
      default: "mini"
    }
  },
  data() {
    return {
      searchCondition: "", // 下拉项的过滤条件
      busKeywords: "" // 条件输入框绑定值
    };
  },
  computed: {
    bindValue: {
      get() {
        return this.value;
      },
      set(val) {
        this.$emit("update:value", val);
      }
    },
    // 下拉项绑定值
    filterOptions() {
      if (this.searchCondition) {
        return this.groupCircleOpts.filter(item => item.name.includes(this.searchCondition));
      } else {
        return this.groupCircleOpts;
      }
    }
  },
  methods: {
    focusSelect() {
      this.clearSearch()
      this.$emit("focusSelect");
    },
    clearSearch() {
      this.busKeywords = "";
      this.searchCondition = "";
    }
  }
};
</script>
```
